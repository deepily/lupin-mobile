# LUPIN MOBILE - FLUTTER UI SPECIFICATION

**Date**: 2025.07.07  
**Status**: UI Design Phase  
**Priority**: High - Core Application Interface  
**Source**: Web UI Analysis from queue.html and JavaScript components  

## Executive Summary

This document defines the Flutter mobile UI specification based on analysis of the existing web-based Lupin application. The web app serves as a job queue management system with real-time communication, TTS capabilities, and user authentication. The mobile app will replicate this functionality with a mobile-first design approach.

## Web Application Analysis

### Core Web UI Components Identified

#### 1. **Main Dashboard (queue.html)**
- **Purpose**: Central hub for job management and system interaction
- **Layout**: Multi-section dashboard with real-time updates
- **Key Features**: Job tracking, TTS controls, authentication, question submission

#### 2. **JavaScript Architecture**
- **hybrid-tts.js**: WebSocket-based TTS streaming with fallback mechanisms
- **queue.js**: Main application logic, WebSocket management, UI updates
- **job-completion-cache.js**: Caching system for completed jobs

#### 3. **Functional Areas**
1. **Job Queue Management**: Track and display job states
2. **Real-time Communication**: WebSocket for live updates
3. **Audio System**: TTS playback and audio notifications
4. **User Interface**: Interactive controls and status displays
5. **Authentication**: Email-based user session management

```dart
class AuthenticationScreen extends StatefulWidget {
  @override
  _AuthenticationScreenState createState() => _AuthenticationScreenState();

class _AuthenticationScreenState extends State<AuthenticationScreen> {
  final TextEditingController _emailController = TextEditingController();
  final AuthenticationBloc _authBloc = GetIt.instance<AuthenticationBloc>();
  final WebSocketService _wsService = GetIt.instance<WebSocketService>();
  
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: SafeArea(
        child: Padding(
          padding: EdgeInsets.all(24.0),
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              // App Logo
              Image.asset('assets/images/lupin_logo.png', height: 120),
              SizedBox(height: 48),
              
              // Title
              Text(
                'LUPIN MOBILE',
                style: Theme.of(context).textTheme.headlineMedium?.copyWith(
                  fontWeight: FontWeight.bold,
                ),
              ),
              SizedBox(height: 48),
              
              // Email Input
              BlocBuilder<AuthenticationBloc, AuthenticationState>(
                builder: (context, state) {
                  return TextFormField(
                    controller: _emailController,
                    keyboardType: TextInputType.emailAddress,
                    decoration: InputDecoration(
                      labelText: 'Email Address',
                      border: OutlineInputBorder(),
                      prefixIcon: Icon(Icons.email),
                      enabled: state is! AuthenticationLoading,
                    ),
                    validator: (value) {
                      if (value?.isEmpty ?? true) return 'Email is required';
                      if (!value!.contains('@')) return 'Invalid email format';
                      return null;
                    },
                  );
                },
              ),
              SizedBox(height: 24),
              
              // Connect Button
              BlocConsumer<AuthenticationBloc, AuthenticationState>(
                listener: (context, state) {
                  if (state is AuthenticationSuccess) {
                    Navigator.pushReplacementNamed(context, '/dashboard');
                  } else if (state is AuthenticationFailure) {
                    ScaffoldMessenger.of(context).showSnackBar(
                      SnackBar(content: Text(state.error)),
                    );
                  }
                },
                builder: (context, state) {
                  return SizedBox(
                    width: double.infinity,
                    height: 56,
                    child: ElevatedButton(
                      onPressed: state is AuthenticationLoading
                          ? null
                          : () => _handleConnect(),
                      child: state is AuthenticationLoading
                          ? CircularProgressIndicator(color: Colors.white)
                          : Text(
                              'Connect',
                              style: TextStyle(fontSize: 18, fontWeight: FontWeight.w600),
                            ),
                    ),
                  );
                },
              ),
              SizedBox(height: 32),
              
              // Connection Status
              BlocBuilder<AuthenticationBloc, AuthenticationState>(
                builder: (context, state) {
                  return Column(
                    children: [
                      Row(
                        mainAxisAlignment: MainAxisAlignment.center,
                        children: [
                          Icon(
                            Icons.circle,
                            size: 12,
                            color: _getConnectionStatusColor(state),
                          ),
                          SizedBox(width: 8),
                          Text(
                            'Connection Status: ${_getConnectionStatusText(state)}',
                            style: TextStyle(fontSize: 14, color: Colors.grey[600]),
                          ),
                        ],
                      ),
                      if (state is AuthenticationSuccess) ...[
                        SizedBox(height: 8),
                        Text(
                          'Session ID: ${state.sessionId}',
                          style: TextStyle(fontSize: 12, color: Colors.grey[500]),
                        ),
                      ],
                    ],
                  );
                },
              ),
            ],
          ),
        ),
      ),
    );
  }
  
  void _handleConnect() {
    final email = _emailController.text.trim();
    if (email.isNotEmpty && email.contains('@')) {
      _authBloc.add(AuthenticationRequested(email: email));
    }
  }
  
  Color _getConnectionStatusColor(AuthenticationState state) {
    if (state is AuthenticationSuccess) return Colors.green;
    if (state is AuthenticationFailure) return Colors.red;
    if (state is AuthenticationLoading) return Colors.orange;
    return Colors.grey;
  }
  
  String _getConnectionStatusText(AuthenticationState state) {
    if (state is AuthenticationSuccess) return 'Connected';
    if (state is AuthenticationFailure) return 'Failed';
    if (state is AuthenticationLoading) return 'Connecting...';
    return 'Disconnected';
  }
}
```

```dart
            _getStatusIcon(status),

          crossAxisAlignment: CrossAxisAlignment.start,
          children: [

                style: TextStyle(fontSize: 12),

            Text(
              'Updated: ${_formatTime(job.updatedAt)}',

            ),

          mainAxisSize: MainAxisSize.min,
          children: [
            if (status == JobStatus.completed || status == JobStatus.dead) ...[
              IconButton(
                icon: Icon(Icons.replay),
                onPressed: () => _replayJob(job),
                tooltip: 'Replay Job',
              ),
            ],
            if (status == JobStatus.running) ...[
              IconButton(
                icon: Icon(Icons.pause),
                onPressed: () => _pauseJob(job),
                tooltip: 'Pause Job',
              ),
            ],
            IconButton(
              icon: Icon(Icons.visibility),
              onPressed: () => _viewJobDetails(job),
              tooltip: 'View Details',
            ),
          ],
        ),
        onTap: () => _viewJobDetails(job),
      ),
    );
  }
  
  Widget _buildAudioControls() {
    return Container(
      padding: EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: Theme.of(context).cardColor,
        boxShadow: [
          BoxShadow(
            color: Colors.black.withOpacity(0.1),
            blurRadius: 4,
            offset: Offset(0, -2),
          ),
        ],
      ),
      child: Row(
        children: [
          Expanded(
            child: ElevatedButton.icon(
              onPressed: () => _playAllQueued(),
              icon: Icon(Icons.play_arrow),
              label: Text('Play All'),
            ),
          ),
          SizedBox(width: 12),
          BlocBuilder<TTSBloc, TTSState>(
            builder: (context, state) {
              return Row(
                children: [
                  Text('Auto TTS:'),
                  SizedBox(width: 8),
                  Switch(
                    value: state.autoPlayEnabled,
                    onChanged: (value) => _toggleAutoTTS(value),
                  ),
                ],
              );
            },
          ),
          SizedBox(width: 12),
          IconButton(
            icon: Icon(Icons.volume_up),
            onPressed: () => _showVolumeControls(),
          ),
        ],
      ),
    );
  }
  
  // Helper Methods
  IconData _getStatusIcon(JobStatus status) {
    switch (status) {
      case JobStatus.todo:
        return Icons.schedule;
      case JobStatus.running:
        return Icons.play_circle;
      case JobStatus.completed:
        return Icons.check_circle;
      case JobStatus.dead:
        return Icons.error;
    }
  }
  
  Color _getStatusColor(JobStatus status) {
    switch (status) {
      case JobStatus.todo:
        return Colors.orange;
      case JobStatus.running:
        return Colors.blue;
      case JobStatus.completed:
        return Colors.green;
      case JobStatus.dead:
        return Colors.red;
    }
  }
  
  String _formatTime(DateTime dateTime) {
    final now = DateTime.now();
    final difference = now.difference(dateTime);
    
    if (difference.inMinutes < 1) {
      return 'Just now';
    } else if (difference.inHours < 1) {
      return '${difference.inMinutes}m ago';
    } else if (difference.inDays < 1) {
      return '${difference.inHours}h ago';
    } else {
      return '${difference.inDays}d ago';
    }
  }
  
  // Action Methods
  void _replayJob(Job job) {

    _queueBloc.add(JobPauseRequested(jobId: job.id));

    _ttsService.playAllQueued();

  void _toggleAutoTTS(bool enabled) {

  }
  
  void _showAudioControls() {
    showModalBottomSheet(
      context: context,
      builder: (context) => AudioControlsSheet(),
    );
  }
  
  void _showVolumeControls() {
    // Implementation for volume controls
  }
```

```dart
    with TickerProviderStateMixin {
  final VoiceInputBloc _voiceBloc = GetIt.instance<VoiceInputBloc>();
  final TextEditingController _textController = TextEditingController();
  late AnimationController _animationController;
  late Animation<double> _scaleAnimation;
  
  @override
  void initState() {
    super.initState();
    _animationController = AnimationController(
      duration: Duration(seconds: 1),
      vsync: this,
    );
    _scaleAnimation = Tween<double>(
      begin: 1.0,
      end: 1.2,
    ).animate(CurvedAnimation(
      parent: _animationController,
      curve: Curves.easeInOut,
    ));
  }
  
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Voice Input'),
        actions: [
          IconButton(
            icon: Icon(Icons.volume_up),
            onPressed: () => _showAudioSettings(),
          ),
        ],
      ),
      body: BlocConsumer<VoiceInputBloc, VoiceInputState>(
        listener: (context, state) {
          if (state is VoiceInputRecording) {
            _animationController.repeat(reverse: true);
          } else {
            _animationController.stop();
            _animationController.reset();
          }
          
          if (state is VoiceInputTranscribed) {
            _textController.text = state.transcription;
          }
          
          if (state is VoiceInputSubmitted) {
            Navigator.pop(context);
            ScaffoldMessenger.of(context).showSnackBar(
              SnackBar(content: Text('Question submitted successfully!')),
            );
          }

                  child: AnimatedBuilder(
                    animation: _scaleAnimation,
```

```dart
// Events
abstract class AuthenticationEvent {}
class AuthenticationRequested extends AuthenticationEvent {
  final String email;
  AuthenticationRequested({required this.email});
}
class AuthenticationLogoutRequested extends AuthenticationEvent {}

// States
abstract class AuthenticationState {}
class AuthenticationInitial extends AuthenticationState {}
class AuthenticationLoading extends AuthenticationState {}
class AuthenticationSuccess extends AuthenticationState {
  final String email;
  final String sessionId;
  final String token;
  AuthenticationSuccess({required this.email, required this.sessionId, required this.token});
}
class AuthenticationFailure extends AuthenticationState {
  final String error;
  AuthenticationFailure({required this.error});
}

// BLoC
class AuthenticationBloc extends Bloc<AuthenticationEvent, AuthenticationState> {
  final WebSocketService _webSocketService;
  final AuthRepository _authRepository;
  
  AuthenticationBloc({
    required WebSocketService webSocketService,
    required AuthRepository authRepository,
  }) : _webSocketService = webSocketService,
       _authRepository = authRepository,
       super(AuthenticationInitial()) {
    
    on<AuthenticationRequested>(_onAuthenticationRequested);
    on<AuthenticationLogoutRequested>(_onAuthenticationLogoutRequested);
  }
  
  Future<void> _onAuthenticationRequested(
    AuthenticationRequested event,
    Emitter<AuthenticationState> emit,
  ) async {
    emit(AuthenticationLoading());
    
    try {
      // Generate session ID and token
      final sessionId = _generateSessionId();
      final token = _generateToken(event.email);
      
      // Establish WebSocket connection
      await _webSocketService.connect(
        sessionId: sessionId,
        token: token,
      );
      
      // Save authentication data
      await _authRepository.saveAuthData(
        email: event.email,
        sessionId: sessionId,
        token: token,
      );
      
      emit(AuthenticationSuccess(
        email: event.email,
        sessionId: sessionId,
        token: token,
      ));
    } catch (error) {
      emit(AuthenticationFailure(error: error.toString()));
    }
  }
  
  String _generateSessionId() {
    return uuid.v4();
  }
  
  String _generateToken(String email) {
    // Simple token generation - in production use proper JWT
    return base64Encode(utf8.encode('${email}_${DateTime.now().millisecondsSinceEpoch}'));
  }
}
```

```dart
// BLoC
class QueueBloc extends Bloc<QueueEvent, QueueState> {
  final WebSocketService _webSocketService;
  final QueueRepository _queueRepository;
  StreamSubscription<dynamic>? _webSocketSubscription;
  
  QueueBloc({
    required WebSocketService webSocketService,
    required QueueRepository queueRepository,
  }) : _webSocketService = webSocketService,
       _queueRepository = queueRepository,
       super(QueueInitial()) {
    
    on<QueueStarted>(_onQueueStarted);
    on<QueueUpdated>(_onQueueUpdated);
    on<JobReplayRequested>(_onJobReplayRequested);
    on<JobPauseRequested>(_onJobPauseRequested);
  }
  
  Future<void> _onQueueStarted(
    QueueStarted event,
    Emitter<QueueState> emit,
  ) async {
    emit(QueueLoading());
    
    try {
      // Load initial queue data
      final initialJobs = await _queueRepository.getAllJobs();
      _emitJobsState(initialJobs, emit);
      
      // Subscribe to WebSocket updates
      _webSocketSubscription = _webSocketService.stream.listen(
        (message) => _handleWebSocketMessage(message),
      );
    } catch (error) {
      // Handle error
    }
  }
  
  void _handleWebSocketMessage(dynamic message) {
    if (message['type'] == 'queue_update') {
      final jobs = (message['jobs'] as List)
          .map((json) => Job.fromJson(json))
          .toList();
      add(QueueUpdated(jobs: jobs));
    }
  }
  
  void _emitJobsState(List<Job> jobs, Emitter<QueueState> emit) {
    final todoJobs = jobs.where((job) => job.status == JobStatus.todo).toList();
    final runningJobs = jobs.where((job) => job.status == JobStatus.running).toList();
    final completedJobs = jobs.where((job) => job.status == JobStatus.completed).toList();
    final deadJobs = jobs.where((job) => job.status == JobStatus.dead).toList();
    
    emit(QueueLoaded(
      todoJobs: todoJobs,
      runningJobs: runningJobs,
      completedJobs: completedJobs,
      deadJobs: deadJobs,
    ));
  }
}
```

### WebSocket Service
```dart
class WebSocketService {
  IOWebSocketChannel? _channel;
  String? _sessionId;
  String? _token;
  final StreamController<dynamic> _messageController = StreamController.broadcast();
  
  Stream<dynamic> get stream => _messageController.stream;
  
  Future<void> connect({
    required String sessionId,
    required String token,
  }) async {
    try {
      _sessionId = sessionId;
      _token = token;
      
      final uri = Uri.parse('ws://localhost:7999/ws/queue/$sessionId');
      _channel = IOWebSocketChannel.connect(uri);
      
      // Send authentication
      _channel!.sink.add(jsonEncode({
        'type': 'auth',
        'token': token,
      }));
      
      // Listen to messages
      _channel!.stream.listen(
        _handleMessage,
        onError: _handleError,
        onDone: _handleDisconnection,
      );
    } catch (error) {
      throw WebSocketException('Failed to connect: $error');
    }
  }
  
  void _handleMessage(dynamic message) {
    try {
      final decoded = jsonDecode(message as String);
      _messageController.add(decoded);
    } catch (error) {
      print('Error decoding message: $error');
    }
  }
  
  void _handleError(dynamic error) {
    print('WebSocket error: $error');
    _messageController.addError(error);
  }
  
  void _handleDisconnection() {
    print('WebSocket disconnected');
    // Implement reconnection logic
  }
  
  void sendMessage(Map<String, dynamic> message) {
    if (_channel != null) {
      _channel!.sink.add(jsonEncode(message));
    }
  }
  
  void disconnect() {
    _channel?.sink.close();
    _channel = null;
  }
}
```

### TTS Service Integration
```dart
class TTSService {
  final WebSocketService _webSocketService;
  final AudioPlayer _audioPlayer = AudioPlayer();
  final Queue<AudioNotification> _audioQueue = Queue();
  bool _isPlaying = false;
  bool _autoPlayEnabled = true;
  
  TTSService({required WebSocketService webSocketService})
      : _webSocketService = webSocketService;
  
  Future<void> synthesizeText(String text) async {
    try {
      // Send TTS request via WebSocket
      _webSocketService.sendMessage({
        'type': 'tts_request',
        'text': text,
        'voice_id': 'default',
        'options': {
          'streaming': true,
          'format': 'mp3',
        },
      });
    } catch (error) {
      throw TTSException('Failed to synthesize text: $error');
    }
  }
  
  void addToQueue(AudioNotification notification) {
    _audioQueue.add(notification);
    if (_autoPlayEnabled && !_isPlaying) {
      _playNext();
    }
  }
  
  Future<void> _playNext() async {
    if (_audioQueue.isEmpty || _isPlaying) return;
    
    _isPlaying = true;
    final notification = _audioQueue.removeFirst();
    
    try {
      await synthesizeText(notification.text);
      // Audio will be played via WebSocket stream
    } catch (error) {
      print('Error playing audio: $error');
    } finally {
      _isPlaying = false;
      // Play next in queue
      if (_audioQueue.isNotEmpty) {
        _playNext();
      }
    }
  }
  
  void playAllQueued() {
    if (!_isPlaying && _audioQueue.isNotEmpty) {
      _playNext();
    }
  }
  
  void setAutoPlay(bool enabled) {
    _autoPlayEnabled = enabled;
    if (enabled && !_isPlaying && _audioQueue.isNotEmpty) {
      _playNext();
    }
  }
  
  void clearQueue() {
    _audioQueue.clear();
    _audioPlayer.stop();
    _isPlaying = false;
  }
}
```
